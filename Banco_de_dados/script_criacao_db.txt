CREATE DATABASE sistema_wishbank;
USE sistema_wishbank

CREATE TABLE usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    senha VARCHAR(255) NOT NULL,
);

CREATE TABLE registro_golpes (
    id_registro_golpe INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    descricao TEXT,
    status ENUM('Aberto', 'Em Andamento', 'Fechado') DEFAULT 'Aberto',
    data_abertura_reclamacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    data_resposta_reclamacao TIMESTAMP NULL,
    id_usuario INT NOT NULL,
    numero_golpe VARCHAR(150) NOT NULL
    qtd_golpes INT NOT NULL
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON UPDATE CASCADE

);
